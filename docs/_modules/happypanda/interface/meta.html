[docs]def get_version():
    """
    Get version of components: 'core', 'db' and 'torrent'

    Returns:
        a dict of component: list of major, minor, patch
    """
    vs = dict(
        core=list(constants.version),
        db=list(constants.version_db),
        torrent=(0, 0, 0)
    )
    return message.Identity("version", vs)
